\section{cbf}

%Le control barrier function traggono ispirazione dalle control Lyapunov function , che rappresentano l'estensione al controllo delle funzioni di Lyapunov . Allo stesso modo le CBF sono l'estensione al controllo della funzioni di barriera , utilizzate in ottimizzazione.  Lo scopo principale delle CBF è definire un sottoinsieme  invariante in avanti , ossia un insieme per cui ogni traiettoria che parte da esso rimane al suo interno per ogni tempo futuro . Inoltre sono molto utili nella sintesi di controllori che ci permettono di rimanere nel sottoinsieme. \\
%Consideriamo il sistema dinamico non lineare affine al controllo \\

Control barrier function are inspired by the control Lyapunov function , which are the extension to control theory  of the Lyapunov function. In the same way CBF are the extension to contro theory  of the barrier function , used in optimization . The main objective of CBF is to define a forward invariant subset , a subset for which every trajectory that start in the subset is confined in there for all future times . CBF are really useful in the synthesis of controls that let us stay in the foward invariant subset .  \\
Let us consider the non-linear control affine system 
\begin{equation}
	\label{eq:2.2.1}
	\dot{x} = f(x) + g(x) u 
\end{equation}
where $x \in \R^n \ , \ u \in \R^m$ are  respectively  the state  and the input of the system. The function  $f : \R^n \rightarrow \R^n$ e $g : \R^n \rightarrow \R^{n \times m}$ are locally Lipschitz  . \\
Let us consider the set   $S$  , defined  as the  superlevel set  of a continuously differentiable function $h : \R^n \rightarrow \R$ : 
\begin{align}
	\label{eq:2.2.2}
 	S &= \{ x \in \R^n | h(x) \geq 0 \} \\
	S^0 &= \{x \in \R^n \ | \ h(x) > 0 \} \\
	\label{eq:2.2.3}
	\partial S & = \{x \in \R^n \ | \ h(x)=0\}
\end{align}
\begin{definizione}
The set $S$ is forward invariant if for every $x_0 \in S , x(t) \in S $ for $x(0)=x_0$ and all $t \in I(x_0)$. The system \ref{eq:2.2.1} is safe with respect to the set $S$ if $S$ is foward invariant  
\end{definizione}
\begin{definizione}
The $h \in \C^1(D) , h : D \subseteq \R^n \rightarrow \R$ is control barrier function , for the system  \ref{eq:2.2.1} , on the set  S , defined by \ref{eq:2.2.2} - \ref{eq:2.2.3} , if  there exists an extended class  $\kappa_{\infty}$ function $\alpha$  
%\footnote{Per $a \in R^+$ , una funzione continua $\alpha : [0 ,a ) \rightarrow \R$ è una funzione di classe $\kappa$ se è strettamente crescente e $\alpha(0)=0$. Una funzione continua $\alpha : (-b ,a ) \rightarrow \R , \ \ \ a , b \in \R^+$  è una funzione di classe $\kappa$ estesa se è strettamente crescente e $\alpha(0)=0$. Inoltre , se $a = b = \infty$ e $\lim_{r \rightarrow \infty } \alpha(r)= \infty $ , allora viene chiamata funzione di classe $\kappa_{\infty} $ estesa } 
\footnote{For $a \in \R^+$, a continuous function $\alpha: [0, a) \rightarrow \R$ is a class $\kappa$ function if it is strictly increasing and $\alpha(0) = 0$. A continuous function $\alpha: (-b, a) \rightarrow \R$, with $a, b \in \R^+$, is an extended class $\kappa$ function if it is strictly increasing and $\alpha(0) = 0$. Furthermore, if $a = b = \infty$ and $\lim_{r \rightarrow \infty} \alpha(r) = \infty$, then it is called an extended class $\kappa_\infty$ function.}
such that : 
\begin{align}
	\sup_{u \in \R^m} &[L_f h(x) + L_g h(x) u \geq - \alpha (h(x))]  \ \   \forall x \in S \notag  \\ 
	\sup_{u \in \R^m} &[\grad{h(x)} \cdot f(x) + \grad{h(x)}^T  g(x) u \geq - \alpha (h(x))]\ \ \   \forall x \in S  \label{eq:2.5}
\end{align}
where 
\[L_g h = \begin{bmatrix}
	L_{g_1} h \\
	L_{g_2} h \\
	\vdots \\
	L_{g_m} h 
\end{bmatrix}^T\] so $g_i$ is the i column of g 
 \end{definizione}
 Let us consider the set of all the control that render the set $S$ forward invariant :
 $$K_{cbf}(u)= \{u \in \R^m \ | \ L_f h(x) + L_g h(x) u \geq - \alpha (h(x)) \}$$
 \begin{teo}{}{}
 	Let $S \subset  \R^n$ be a set defined as the superlevel set of  a continuously differentiable function $h:D \subset  \R^n \rightarrow \R $ , if h is a control barrier function on D and $\grad{h(x)} \neq 0 \ \ \forall x \in \partial S  $ , the any Lipschitz continuous controller  $u(x) \in K_{cbf}$ for the system  \ref{eq:2.2.1} renders the set  S forward invariant  
 \end{teo}
 \begin{proof}
 	For all $x \in \partial S $ we have $L_f h(x) + L_g h(x) u \geq - \alpha (h(x)) = 0 $.Then we can define the contingent cone $T_S(x) = \{v \in \R^n \ | \ \grad{h(x)\cdot v} \geq 0\}$. Wee know that $\forall x \in S^0  \ \ T_S(x)=\R^n$ , let's what happens for the boundary points 
 \begin{align*}
	\grad{h(x)}(f(x)+g(x)u) \geq 0  \\
	\grad{h(x)} f(x) + \grad{h(x)} g(x) u  \geq 0 \\  
L_f h(x) + L_g h(x) u  \geq 0 
 \end{align*}
 then S is  viability domain and then S is also  locally viable under f. \\ 
 We are working under the hypothesis that $ f , g , u$ are locally Lipschitz function so the Cauchy-Lipschitz Theorem guarantees existence and uniqueness of the solution . Knowing this the condition for invariance and viabilitlity corrispondes , so the set S is forward invariant 
 	 \end{proof}
 Furthermore control barrier function give us a necessary and sufficient condition for forward invariance if the set S is compact
 \begin{teo}{}{}
 	Let  S be the compact superlevel set of the function $h \in C^1 \ , \ h : \R^n \rightarrow \R$ , with $\grad{h(x)}\neq 0 \ \ \forall x \in \partial S$.If there exists a control law  $u=k(x)$ that renders the subset S forward invariant , then $h|_S : S \rightarrow \R$ è is a control barrier function on S
 \end{teo}
 \begin{proof}
 	Let $h \in C^1 , h|_S : S \rightarrow \R $. For all  $r \geq  0 $ the set  $\{x \ | \ 0 \leq h(x) \leq r\}$ is a compact subset of  C . We define the function $\alpha : [0 , \infty )\rightarrow \infty $ with 
 	$$\alpha(r) = - \inf_{\{x \ | \ 0 \leq h(x) \leq r\}} L_f h + L_g h \ u $$ . Using the compactness of the subset and the continuity of  $L_f h + L_g h \ u $ , $\alpha$ is  well defined function , increasing  and  satisfies  $$L_hf + L_g h \ u \geq  - \alpha (h(x))$$ For Nagumo Theorem  \ref{teo:1} the forward invariance  of  C is equivalent to : $$L_hf + L_g h \ u \geq   0  \ \ \ \  \forall x \in \partial S $$ , this implies that $\alpha(0) \leq 0$. We can always choose a function of class $\kappa \ \hat{\alpha }$ che maggiora $\alpha$ , facendo così abbiamo ottenuto  $L_hf + L_g h \ u \geq  -\hat { \alpha } (h(x))$.
 \end{proof}
 Now we know that the CBF give us conditions for the safety of a set , we want to find a method to creates the control laws that achieves safety . \\
 Suppose that we have a control law $u$ for the system  \ref{eq:2.2.1} and we want to make sure that it enforce forward invariance . Let us suppose that $ u \in K_{cbf}$ per some $x \in \R^n$ . We want to modify this controller to ensure forward invariance , we start by noting that the condition \ref{eq:2.5}  is affine in u , then we can consider the quadratic programming problem (QP)
 \begin{align*}
 	u(x) &= \argmin_{u \in U \subseteq \R^n} \frac{1}{2}\norm{u - k(x)}_2^2 \\
 &	s.t \ \ \ [L_f h(x) + L_g h(x) u \geq - \alpha (h(x))
 \end{align*}
For this for this problem there exists a solution  $u^*$ in closed form : 
 \begin{equation}
u^*  = u_{safe} + u_{nom}
\label{eq_star}
 \end{equation} where 
 \begin{equation}
 u_{safe} = \begin{cases}
 	- \frac{\Phi ( x , t , u_{des})}{L_gh(x)L_gh(x)^T} L_gh(x)^T   \ \ \ \ \ \Phi < 0 \\
 	0 \ \ \ \ \ \ \ \Phi \geq  0 
 \end{cases} 
 \label{eq_safe}
 \end{equation}
 with 
 
 \begin{equation}
\Phi ( x , t , u_{des}) \triangleq \grad{h(x)}^T f(x) + \grad{h(x)}^T g(x) u  +\alpha (h(x))
\label{eq_phi}
 \end{equation}
We now show an example of the usage of CBF
 \begin{es}
 	We want to move a material point from an initial position $q_0 \in \R^2$ to a target position $q_g \in \R^2$ while avoiding a spherical obstacle , with center $q_{obs} \in \R^2 $ and radius $r_{obs}$ . The system is a single integrator : 
 	\[ \dot{q}(t) = u(t) \]
 	then $f(q) = 0 $ , $g(q) =
 	 \begin{bmatrix}
 		1 & 0 \\
 		0 & 1 
 	\end{bmatrix}$ and $ u(t) = \begin{bmatrix}
 	u_1(t) \\
 	u_2(t)
 	\end{bmatrix} $ \\
 	We start by introducing the control barrier function
 	\[h(x) = \norm{q-q_{obs}} - r_{obs}\] which defines the set 

\begin{align*}
S &= \{q \in R^2\ |\  \norm{q-q_g} - r_{obs}  \geq 0 \}  \\
& =  \{q \in R^2\ |\  \sqrt{(x-x_{obs})^2+(y-y_{obs})^2} \geq  r_{obs}  \}  
\end{align*}

	\caption{The set S}
\end{figure}
 \end{es}
 We have out nominal control which is 
 $$u_{nom} =  K * \frac{q_g - q}{||q_g-q||} \quad  K \in \R_{>0}$$
by using equation \ref{eq_phi} and using a constant $\alpha \in \R_{>0}$ as the $\kappa_{\infty}$ function we obtain : 
\begin{align*}
	\nabla h &= 
	\begin{bmatrix}
		\frac{x}{(\sqrt{x-x_{obs})^2+(y-y_{obs})^2}} \\
		\frac{t}{\sqrt{(x-x_{obs})^2+(y-y_{obs})^2}}
	\end{bmatrix} \\
	L_g h &= \nabla h ^T \\
	L_gh L_gh^T &=  \frac{x^2+y^2}{(x-x_{obs})^2+(y-y_{obs})^2}\\
	\Phi &=  \frac{x}{(\sqrt{x-x_{obs})^2+(y-y_{obs})^2}} u_1 + \frac{y}{(\sqrt{x-x_{obs})^2+(y-y_{obs})^2}} u_2 + \alpha ( \norm{q-q_{obs}} - r_{obs})
\end{align*}
and now we have $u_{safe}$ and by adding out $u_{nom}$ we can create our controller that achieves obstacle avoidance while going toward the goal .

 \[
 \begin{cases}
 	ml^2 \ddot{\theta} - m \ddot{x} l \cos(\theta) - mgl\sin(\theta) = 0  \\
 	(M+m)\ddot{x} - ml \ddot{\theta}\cos(\theta) +ml\dot{\theta}^2\sin(\theta)= u
 \end{cases}
 \]
 \[
 \begin{cases}
 	\ddot{x} = \frac{u + m \sin(\theta) \left( g \cos(\theta) - l \dot{\theta}^2 \right)}{M + m \sin^2(\theta)} \\[12pt]
 	\ddot{\theta} = \frac{(M + m)g \sin(\theta) - m l \dot{\theta}^2 \sin(\theta) \cos(\theta) + u \cos(\theta)}{l \left( M + m \sin^2(\theta) \right)}
 \end{cases}
 \]

 \subsubsection{Angolo Critico}
 After some calculations we obtain : 
 \[\ddot{\te} = \frac{-m\cos(\te)\sin(\te)\dot{\te}^2}{(M+m\sin(\te)^2} + u\frac{\cos(\te)}{l(M+m\sin(\te)^2)} + \frac{(M+m)g\sin(\theta)}{l(M+m\sin(\te)^2)}\]
We can rewrite by defining the function 
\begin{align*}
	g(\te) &= \frac{\cos(\te)}{l(M+m\sin(\te)^2)} \\
	h(\theta , \dot{\te} , u ) &= lm\dot{\te}^2 - \frac{u + g(M+m)\tan(\te)}{\sin(\te)} \\
	\ddot{\te} &= - \sin(\theta) g(\te)  h(\te,\dot{\te},u) 
\end{align*}
We want to prove that there exist a crtical anglie $\te_c \in ]-\pi/2 , \pi/2[$ . We note that $g(\te) > 0  \quad \forall \te \in ] -\pi/2 , \pi/2 [$ , so if we want the acceleration the have the opposite sign of $\te$ we need to guarantee that $h(\te , \dot{\te} , u) \geq 0 $
\begin{align*}
	lm\dot{\te}^2 - \frac{u + g(M+m)\tan(\te)}{\sin(\te)} \geq 0  \\
	lm\dot{\te}^2 \geq  \frac{u + g(M+m)\tan(\te)}{\sin(\te)}   \\
	\text{Now we take the Infimum} (\dot{\te} = 0 )\\
	\frac{u + g(M+m)\tan(\te)}{\sin(\te)} \leq 0	
\end{align*} 
 We now have two cases
 \begin{itemize}
 \item $\sin(\te) > 0 $ 
 \begin{align*}
 	u_{min} \leq u &\leq -g(M+m)\tan(\te)\\
 	\tan(\te) &\leq \frac{-u_{min}}{g(M+m)}
 \end{align*}
 \item  $\sin(\te) < 0 $ 
  \begin{align*}
 	u_{max} \geq u & \geq -g(M+m)\tan(\te)\\
 	\tan(\te) &\geq \frac{-u_{max}}{g(M+m)}
 \end{align*}
 \end{itemize}
 so we obtain 
 \[\arctan(-\frac{u_{max}}{g(M+m)} )\leq \theta \leq \arctan(-\frac{u_{min}}{g(M+m)} )\]
