\chapter{Control Barrier Function}

\section{Nagumo's Theorem}
We presented Nagumo's Theorem following   \cite{VT} , which is one of the fundamental building blocks for control barrier function
\begin{definizione}
Let K be a subset of a finite dimensional vector-space (or even a normed space) X . We shall say that a function $x (\cdot) : [0,T] \rightarrow X$ is \textit{viable} in K on $[0,T]$ if 
	$$\forall t \in [0,T] \ , \ \ x(t) \in K$$ 
\end{definizione}
Consider 
\begin{equation}
	\label{diff:1}
	\begin{cases}
		\dot{x}(t)= f(x(t)) \ \ \ \forall t \in [0,T ] \\ 
		x(0) = x_0
	\end{cases}
\end{equation}
con $f : \Omega \subset_{\text{op}} X \rightarrow X$ 
\begin{definizione}
	Let K be a subset of  $\Omega$. 
We shall say that K is \textit{ locally viable} under $f$ (or enjoys the local viability propery for the map $f$) if for any initial state  $x_0 \in K $ , there exists $T > 0$ and  a viable solution  su $[0,T]$  to the differential equation \ref{diff:1} with initial condition $x_0$. 
It is said to be globally viable under $f$  if we can always take $T= \infty$ \ 
The subset K is said to be invariant under $f$ if for any initial state $x_0$ of K , all solutions to the differential equation  \ref{diff:1} are viable in K 
\end{definizione}
The concept of invariance depends upon the behavior of f on the domain $\Omega$ outside $K$ . But we observe that viability property depends only on the behavior of $f$ on K 
\begin{definizione}[ Bouligand Contingent cone]
	Let X be a normed space , K be a nonempty subset of X and x belong to K . The contingent cone to K at x is the set
	$$T_K(x)= \{v \in X \ | \ \liminf_{h\rightarrow 0^+ }\frac{d_K(x+hv)}{h}=0\}$$
	where $d_k(x):= \inf_{y \in K } \norm{x-y}$
\end{definizione}
In other worlds , v belongs to
$T_K(x)$ if and only if exist a sequence of  $h_n >0$ $h_n \rightarrow 0^+$ and a sequence $v_n \in X$ , $v_n \rightarrow v $ such that
$$\forall n \in \mathbb{N} \ , \ x+h_nv_n \in K $$

We see that  $\forall x \in Int(K) , \quad T_K(x)=X$. Therefore , when K is open , the contingent cone to K at any point $x in K$ is always equal to the whole space. Therefore we will be mostly interested in defining the contingent cone for point in the boundary of K . In any case the contingent cone of a point x in the boundary can be the whole space

\begin{lemma}
Let  $x: [0,T] \rightarrow K $ be a differentiable viable function 
	$$\forall t \in [0,T) \ \ x'(t) \in T_K(x)$$
\end{lemma}
\begin{definizione}[Viability Domain]
	Let K be a subset of  $\Omega$. We shall say that K is a viability domain of the map  $f : \Omega \rightarrow X$ se 
	$$\forall x \in X  \ \ , \ \ f(x) \in T_K(x)$$
\end{definizione}
\begin{teo}{Nagumo}{}
	\label{teo:1}
	Let us assume that 
	\begin{enumerate}
		\item K is locally compact
		\item f is continuous from K to X 
	\end{enumerate} 
	The K is locally viable under f if and only if K is viability domain of f 
\end{teo}
\begin{es}
	Let $K \subset \R$, with $K = [-2, 2]$, and $\dot{x}(t) = x(t)$. $K$ is a bounded and closed interval, therefore $K$ is compact. Moreover, the function is continuous, therefore the hypotheses of the theorem are satisfied. Let's check whether $K$ is a \textit{viability domain}. We know that if $x \in (-2, 2)$, then $T_K(x) = \R$. Now let's analyze the two boundary points:
	
	\begin{itemize}
		\item If $x = 2 \rightarrow f(x) = 2$. The tangent cone represents the directions in which I can move without exiting $K$. In this case, I can only move to the left, so I obtain $T_K(2) = (-\infty, 0]$.
		\item Following a reasoning similar to the previous one, We obtain $T_K(-2) = [0, +\infty)$.
	\end{itemize}
	
	We can see that for both boundary points we obtain $f(x) \notin T_K(x)$, therefore $K$ is not a viability domain for $f$.
	
	Now, let's consider the interval $K = (-\infty, 0]$. In this case, the subset is not compact but locally compact, since it is a closed subset of a locally compact space, the Euclidean space. As before, if we take $x \in (-\infty, 0)$, we obtain $T_K(x) = \R$. Let's now move on to study the point $x = 0$, for which we obtain $f(0) = 0$; instead, the tangent cone at the point zero is $T_K(0) = (-\infty, 0]$. We can observe that $f(0) \in T_K(0)$, therefore $K$ is a viability domain, and hence locally viable.
\end{es}

\begin{es}
	\label{es:1.2}
	Let $K \subset \R^2$ be given by $K = \{(x,y) \in \R^2 \mid x^2 + y^2 - 5 \leq 0\} = \{(x,y) \in \R^2 \mid g(x,y) \leq 0\}$, and let us consider the vector field $f(x,y) = (x, y^2)$. $K$ is a compact subset and $f$ is a continuous vector field. 
	
	We know that for interior points $x \in K^\circ$, the tangent cone is $T_K(x) = \R^2$. Let us now consider the boundary points $x \in \partial K$, which belong to the level curve $g^{-1}(0) = \{(x,y) \in \R^2 \mid x^2 + y^2 = 5\}$. In this case, the boundary curve is differentiable, and we can define the tangent cone using tangent lines. Indeed,
	$$T_K(x_0) = \{v \in \R^2 \mid g(x_0) + \nabla g(x_0) \cdot v \leq 0 \}$$
	%Alternatively, using the fact that the gradient of $g$ is always orthogonal to its level curves and points in the direction of maximum increase, we have:
	$$T_K(x_0)  = \{v \in \R^2 \mid \nabla g(x_0) \cdot v \leq 0 \}$$
	
	Let us compute the condition for the vector field to belong to the tangent cone at the boundary:
	\begin{align*}
		\nabla g(x) \cdot f(x) &= (2x, 2y) \cdot (x, y^2) \\
		&= 2x^2 + 2y^3 \leq 0 \\
		&\implies y \leq -x^{\frac{2}{3}}
	\end{align*}
	
	\begin{figure}[h]
		\centering
		\includegraphics[width=0.8\linewidth]{immagini/circonferenza}
		\caption{Example \ref{es:1.2}(a)}
		\label{fig:circonferenza} 
	\end{figure}
	
	Indeed, evaluating this at the boundary point $x_0 = (0, \sqrt{5})$:
	\begin{align*}
		f(x_0) &= (0, 5) \\
		g(x_0) &= 0 \\
		\nabla g(x_0) &= (0, 2\sqrt{5}) \\
		\nabla g(x_0) \cdot f(x_0) &= 10\sqrt{5} > 0 \implies f(x_0) \notin T_K(x_0) 
	\end{align*}
	We have thus shown that $K$ is not a viability domain for $f$, meaning $K$ cannot be locally viable under $f$. \\
	
	Now, consider the same set $K$ but with the vector field $f(x,y) = (-x, -y^3)$. Let us check if this vector field satisfies the condition for belonging to the tangent cone:
	\begin{align*}
		\nabla g(x) \cdot f(x) &= (2x, 2y) \cdot (-x, -y^3)^T \\
		&= -2x^2 - 2y^4 \leq 0 \quad \forall (x,y) \in \R^2 
	\end{align*}
	We have therefore shown that $f(x) \in T_K(x)$ for all $x \in K$. Thus, $K$ is a viability domain for $f$, and by Nagumo's Theorem, $K$ is locally viable under $f$.

		\caption{Example \ref{es:1.2} (b)}
		\label{fig:circonferenza_1} 
	\end{figure}
\end{es}

\begin{teo}{Viability}{}
	Let us consider a closed subset K  of a finite dimensional vector-space X and a continuous map   $f : K \rightarrow X$. \\ If K is a viability domain , then for every initial state $x_0\in K $ there exists a positive T and a viable solution on $[0,T]$ to the differential equation \ref{diff:1} starting at  $x_0$ such that
	$$\begin{cases}
		T= + \infty  \\ 
		T < +\infty \ \ \ \  and \ \ \ \  \limsup_{t\rightarrow T_-} \norm{x(t)}= \infty 
	\end{cases}$$
\end{teo}
Further adequate information - a priori estimates on the growth of $f$  - allows us to exclude the case when  $\lim_{ t \rightarrow T^-} \norm{x(t)}= \infty$.
\begin{teo}{}{}
	Let us consider a subset of K of a finite dimensional vector-space X and a map $f$ form K to X . We assume that the map $f$ is continuous from K to  X , that 
	$$\exists c > 0 \ t.c \ \forall x \in K \ , \ \norm{f(x)} \leq c (\norm{x}+1)$$
	and that K is a closed viability domain of f . \\ Then K is viable under $f$ : for every initial state $x_0 \in K $ , there exists a viable solution on $[0 ,\infty] $ to the differential equation \ref{diff:1} at $x_0$
\end{teo}
